    <script>
        const user = "markdsouza993-creator";
        const repo = "photography-portfolio";
        const folder = "latestimages";

        function loadView(viewName) {
            document.getElementById('view-gallery').style.display = 'none';
            document.getElementById('view-about').style.display = 'none';
            document.getElementById('view-contact').style.display = 'none';
            document.getElementById('view-' + viewName).style.display = 'block';
            document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
            event.target.classList.add('active');
        }

        async function fetchPhotos() {
            const container = document.getElementById('photo-grid');
            const url = `https://api.github.com/repos/${user}/${repo}/contents/${folder}`;
            try {
                const response = await fetch(url);
                if (!response.ok) throw new Error("GitHub Error");
                const data = await response.json();
                const images = data.filter(file => file.name.match(/\.(jpg|jpeg|png)$/i));
                container.innerHTML = "";
                images.forEach(img => {
                    const div = document.createElement('div');
                    div.className = 'grid-item';
                    div.innerHTML = `<img src="${img.download_url}" alt="Photo">`;
                    div.onclick = () => {
                        document.getElementById('lightbox-img').src = img.download_url;
                        document.getElementById('lightbox').style.display = 'flex';
                    };
                    container.appendChild(div);
                });
            } catch (error) { container.innerHTML = "Loading..."; }
        }
        window.onload = fetchPhotos;
    </script>
